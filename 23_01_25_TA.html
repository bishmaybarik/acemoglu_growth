import numpy as np
from sympy import symbols, solve, Interval, Union
import matplotlib.pyplot as plt

# Define the polynomial
k = symbols('k')
polynomial = 4*k**3 - 18*k**2 + 22*k - 6

# Find roots of the polynomial
roots = solve(polynomial, k)
print("Roots of the polynomial:", roots)

# Sort roots for interval analysis
roots = sorted([float(root.evalf()) for root in roots])

# Check intervals where the polynomial is negative
intervals = []
all_intervals = [(-np.inf, roots[0])] + [(roots[i], roots[i+1]) for i in range(len(roots)-1)] + [(roots[-1], np.inf)]

for interval in all_intervals:
    # Pick a test point in the interval
    test_point = (interval[0] + interval[1]) / 2 if interval[1] != np.inf else interval[0] + 1
    if polynomial.subs(k, test_point) < 0:
        intervals.append(interval)

# Display intervals where the polynomial is negative
print("Intervals where the polynomial is negative:", intervals)

# Plot the polynomial
x_vals = np.linspace(min(roots) - 1, max(roots) + 1, 1000)
y_vals = [4*x**3 - 18*x**2 + 22*x - 6 for x in x_vals]

plt.figure(figsize=(10, 6))
plt.plot(x_vals, y_vals, label="4k^3 - 18k^2 + 22k - 6", color='blue')
plt.axhline(0, color='black', linewidth=0.8, linestyle='--')

# Highlight intervals where the polynomial is negative
for interval in intervals:
    if interval[1] == np.inf:
        x_interval = np.linspace(interval[0], interval[0] + 1, 500)
    elif interval[0] == -np.inf:
        x_interval = np.linspace(interval[1] - 1, interval[1], 500)
    else:
        x_interval = np.linspace(interval[0], interval[1], 500)
    y_interval = [4*x**3 - 18*x**2 + 22*x - 6 for x in x_interval]
    plt.fill_between(x_interval, y_interval, 0, where=[y < 0 for y in y_interval], color='red', alpha=0.5)

# Add labels and legend
plt.title("Graph of the First Derivative [f'(k)] and Negative Intervals")
plt.xlabel("k")
plt.ylabel("f(k)")
plt.legend()
plt.grid()
plt.show()

import numpy as np
import matplotlib.pyplot as plt

# Define the second function
def g(k):
    return 4*k**3 - 18*k**2 + 22*k - 6

# Generate values of k for the graph
k_values = np.linspace(-10, 10, 500)  # A wide range for visualization
g_values = g(k_values)

# Plot the function
plt.figure(figsize=(10, 6))
plt.plot(k_values, g_values, label=r"$4k^3 - 18k^2 + 22k - 6$", color="blue")

# Mark the behavior as k approaches 0 and infinity
plt.axhline(0, color='black', linestyle='--', linewidth=0.7)
plt.axvline(0, color='black', linestyle='--', linewidth=0.7)
plt.scatter(0, g(0), color="red", label=f"Limit as $k \\to 0$: {g(0)}")
plt.text(1, g(0) + 10, f"$g(0) = {g(0)}$", fontsize=10, color="red")

# Graph details
plt.title("Graph of $4k^3 - 18k^2 + 22k - 6$", fontsize=14)
plt.xlabel("$k$", fontsize=12)
plt.ylabel("$g(k)$", fontsize=12)
plt.legend(fontsize=12)
plt.grid(alpha=0.5)
plt.show()
